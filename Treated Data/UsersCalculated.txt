USUARIOS - Calidad

IF [difference_last_and_first_share] = -1 THEN 
    "[7] No interesado/No comprendió"
ELSEIF [difference_last_and_first_share] = 0 THEN 
    "[6] No capturado"
ELSEIF [Share Frequency] <= 1 THEN
    IF [difference_last_and_first_share] >= 30 AND [difference_last_and_first_share] <= 60 THEN
        "[0] Diario por una semana" 
    ELSEIF [difference_last_and_first_share] >= 7 AND [difference_last_and_first_share] <= 29 THEN
        "[1] Diario por un mes"
    ELSEIF [difference_last_and_first_share] <= 6 THEN
        "[5] Usuario perdido"
    ELSE
        "[2] Usuario diario constante"
    END
ELSEIF [Share Frequency] <= 7 THEN
    IF [difference_last_and_first_share] >= 30 AND [difference_last_and_first_share] <= 60 THEN
        "[3] Semanal por un mes"
    ELSEIF [difference_last_and_first_share] <= 29 THEN
        "[5] Usuario perdido"
    ELSE
        "[4] Usuario semanal constante"
    END
ELSE
    "[5] Usuario perdido"
END

USUARIOS - Dia

IF [Dia afiliacion] = 0 THEN "Lunes"
ELSEIF [Dia afiliacion] = 1 THEN "Martes"
ELSEIF [Dia afiliacion] = 2 THEN "Miércoles"
ELSEIF [Dia afiliacion] = 3 THEN "Jueves"
ELSEIF [Dia afiliacion] = 4 THEN "Viernes"
ELSEIF [Dia afiliacion] = 5 THEN "Sábado"
ELSE "Domingo"
END

USUARIOS - Edad

DATEDIFF('year',[Nacimiento],DATE("2015-05-02"))

USUARIOS - Shares Frequency

[difference_last_and_first_share]/[Shares Totales]
